let currentType = null;

function fmt(n) { return '₹' + n.toLocaleString('en-IN'); }

function goHome() {
  hide('pg-programs'); hide('pg-electives'); show('pg-landing');
  window.scrollTo(0,0);
}

function showPrograms(type) {
  currentType = type;
  hide('pg-landing'); hide('pg-electives'); show('pg-programs');
  const d = DATA[type];
  document.getElementById('pgtype-label').textContent = type + ' Programs';
  document.getElementById('prog-kicker').textContent = d.kicker;
  document.getElementById('prog-title').innerHTML = type === 'UG'
    ? `Undergraduate <span class="accent">Programs</span>`
    : `Postgraduate <span class="accent">Programs</span>`;
  const progs = Object.keys(d.programs);
  document.getElementById('prog-count-pill').textContent = progs.length + ' Programs Available';
  const grid = document.getElementById('programs-grid');
  grid.innerHTML = '';
  progs.forEach((prog, idx) => {
    const info = d.programs[prog];
    const card = document.createElement('div');
    card.className = 'prog-card';
    card.style.animationDelay = (idx * 0.08) + 's';
    card.innerHTML = `
      <div class="prog-tag">${prog}</div>
      <div class="prog-name">${info.fullName}</div>
      <div class="prog-meta">${type} · ${info.electives.length} Specialization${info.electives.length > 1 ? 's' : ''} Available</div>
      <div class="prog-footer">
        <div class="prog-elective-count"><strong>${info.electives.length}</strong> Specialization${info.electives.length > 1 ? 's' : ''}</div>
        <div class="prog-arrow">→</div>
      </div>`;
    card.onclick = () => showElectives(prog);
    grid.appendChild(card);
  });
  window.scrollTo(0,0);
}

function showElectives(progKey) {
  hide('pg-programs'); hide('pg-landing'); show('pg-electives');
  const d = DATA[currentType];
  const info = d.programs[progKey];
  document.getElementById('bc-pgtype').textContent = currentType + ' Programs';
  document.getElementById('bc-pgtype').onclick = () => showPrograms(currentType);
  document.getElementById('bc-prog').textContent = progKey;
  document.getElementById('elec-kicker').textContent = info.fullName;
  document.getElementById('elec-title').innerHTML = `${progKey} <span class="accent">Specializations</span>`;
  document.getElementById('elec-count-pill').textContent = info.electives.length + ' Specializations';
  document.getElementById('elec-back-btn').onclick = () => showPrograms(currentType);
  const body = document.getElementById('elec-table-body');
  body.innerHTML = '';
  info.electives.forEach((el, i) => {
    const row = document.createElement('div');
    row.className = 'elec-row';
    row.style.animationDelay = (i * 0.06) + 's';
    row.innerHTML = `
      <div class="row-num">${String(i+1).padStart(2,'0')}</div>
      <div class="row-name">${el.name}</div>
      <div class="fee-wrap"><span class="fee-val">${fmt(el.sem)}</span></div>
      <div class="fee-wrap"><span class="fee-val">${fmt(el.yearly)}</span></div>
      <div class="fee-wrap col-total"><span class="fee-val total-val">${fmt(el.total)}</span></div>
      <div class="center col-brochure">
        <a class="dl-btn" href="${el.brochure}" target="_blank" rel="noopener">⬇ Download</a>
      </div>`;
    body.appendChild(row);
  });
  window.scrollTo(0,0);
}
